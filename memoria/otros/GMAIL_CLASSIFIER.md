# 📧 GMAIL CLASSIFIER — SISTEMA DE CLASIFICACIÓN AUTOMÁTICA

## 🎯 OBJETIVO

CONVERTIR GMAIL EN UN SISTEMA ORGANIZADO Y AUTOMATIZABLE, PREPARADO PARA QUE UNA IA PUEDA:

- CLASIFICAR CORREOS AUTOMÁTICAMENTE
- DETECTAR FACTURAS Y RENOVACIONES
- IDENTIFICAR PAGOS Y VENCIMIENTOS
- EXTRAER TAREAS PENDIENTES
- GENERAR RESÚMENES
- FUNCIONAR COMO CENTRO DE CONTROL DIGITAL

---

## 🏗️ ESTRUCTURA DE ETIQUETAS

### 1️⃣ AREA — MACRO CATEGORÍA

| ETIQUETA | DESCRIPCIÓN |
|---|---|
| AREA_PROFESIONAL | CORREOS DE TRABAJO, CLIENTES, PROYECTOS |
| AREA_TECNICA | SERVIDORES, APPS, HERRAMIENTAS, ACCESOS |
| AREA_FINANZAS | FACTURAS, PAGOS, SUSCRIPCIONES, BANCOS |
| AREA_PERSONAL | CORREOS PERSONALES Y FAMILIARES |
| AREA_APRENDIZAJE | CURSOS, NEWSLETTERS, FORMACIÓN |

### 2️⃣ TIPO — NATURALEZA DEL CORREO

| ETIQUETA | DESCRIPCIÓN |
|---|---|
| TIPO_FACTURA | FACTURAS Y RECIBOS |
| TIPO_RENOVACION | AVISOS DE RENOVACIÓN |
| TIPO_SUSCRIPCION | SERVICIOS DE SUSCRIPCIÓN |
| TIPO_CONTRATO | CONTRATOS Y ACUERDOS |
| TIPO_NOTIFICACION | ALERTAS Y NOTIFICACIONES AUTOMÁTICAS |
| TIPO_REUNION | CONVOCATORIAS Y REUNIONES |
| TIPO_LEGAL | AVISOS LEGALES Y PRIVACIDAD |
| TIPO_ACCESO | CREDENCIALES Y ACCESOS |
| TIPO_PROPUESTA | PROPUESTAS Y PRESUPUESTOS |

### 3️⃣ ESTADO — ACCIÓN REQUERIDA

| ETIQUETA | DESCRIPCIÓN |
|---|---|
| ESTADO_PENDIENTE | REQUIERE ACCIÓN |
| ESTADO_ESPERANDO | EN ESPERA DE RESPUESTA |
| ESTADO_RESUELTO | COMPLETADO |

## ⚙️ SCRIPT COMPLETO — APPS SCRIPT

```javascript
// ═══════════════════════════════════════════════════════════════
// GMAIL CLASSIFIER — TODO EN UNO
// 1. CREA LAS ETIQUETAS
// 2. CLASIFICA TODO EL CORREO AUTOMÁTICAMENTE CON CONTINUACIÓN
// EJECUTA SOLO: iniciarSistemaCompleto()
// ═══════════════════════════════════════════════════════════════

const REGLAS = [
  { KEYWORDS: ["FACTURA", "INVOICE", "RECEIPT", "RECIBO", "BILLING STATEMENT"], ETIQUETAS: ["TIPO_FACTURA", "AREA_FINANZAS", "ESTADO_PENDIENTE"] },
  { KEYWORDS: ["RENOVACIÓN", "RENEWAL", "RENEW", "SUBSCRIPTION RENEWING"], ETIQUETAS: ["TIPO_RENOVACION", "AREA_FINANZAS", "ESTADO_PENDIENTE"] },
  { KEYWORDS: ["SUSCRIPCIÓN", "SUBSCRIPTION", "PLAN MENSUAL", "MONTHLY PLAN", "UNSUBSCRIBE"], ETIQUETAS: ["TIPO_SUSCRIPCION", "AREA_FINANZAS"] },
  { KEYWORDS: ["CONTRATO", "CONTRACT", "ACUERDO", "AGREEMENT", "TERMS OF SERVICE"], ETIQUETAS: ["TIPO_CONTRATO", "AREA_PROFESIONAL", "ESTADO_PENDIENTE"] },
  { KEYWORDS: ["REUNIÓN", "MEETING", "CALL SCHEDULED", "ZOOM INVITE", "AGENDA"], ETIQUETAS: ["TIPO_REUNION", "AREA_PROFESIONAL", "ESTADO_PENDIENTE"] },
  { KEYWORDS: ["CONTRASEÑA", "PASSWORD", "ACCESO", "LOGIN", "CREDENTIALS", "2FA"], ETIQUETAS: ["TIPO_ACCESO", "AREA_TECNICA"] },
  { KEYWORDS: ["AVISO LEGAL", "GDPR", "PRIVACY POLICY", "TERMS UPDATED"], ETIQUETAS: ["TIPO_LEGAL", "AREA_FINANZAS"] },
  { KEYWORDS: ["PROPUESTA", "PROPOSAL", "COTIZACIÓN", "QUOTE", "PRESUPUESTO"], ETIQUETAS: ["TIPO_PROPUESTA", "AREA_PROFESIONAL", "ESTADO_ESPERANDO"] },
  { KEYWORDS: ["ALERT", "ALERTA", "ERROR", "DEPLOYMENT", "BUILD FAILED", "SERVER DOWN"], ETIQUETAS: ["TIPO_NOTIFICACION", "AREA_TECNICA", "ESTADO_PENDIENTE"] }
];

// ─── FUNCIÓN PRINCIPAL — EJECUTA ESTA Y SOLO ESTA ───────────────
function iniciarSistemaCompleto() {
  Logger.log("🏗️ PASO 1: CREANDO ETIQUETAS...");
  CREAR_ETIQUETAS();

  Logger.log("🚀 PASO 2: INICIANDO CLASIFICACIÓN DE TODOS LOS CORREOS...");
  PropertiesService.getScriptProperties().setProperty("INICIO", "0");

  ScriptApp.getProjectTriggers().forEach(T => {
    if (T.getHandlerFunction() === "CONTINUAR_CLASIFICACION") {
      ScriptApp.deleteTrigger(T);
    }
  });

  ScriptApp.newTrigger("CONTINUAR_CLASIFICACION")
    .timeBased()
    .everyMinutes(5)
    .create();

  CONTINUAR_CLASIFICACION();
}

// ─── CREA LAS 17 ETIQUETAS ──────────────────────────────────────
function CREAR_ETIQUETAS() {
  const LISTA_ETIQUETAS = [
    "AREA_PROFESIONAL", "AREA_TECNICA", "AREA_FINANZAS", "AREA_PERSONAL", "AREA_APRENDIZAJE",
    "TIPO_FACTURA", "TIPO_RENOVACION", "TIPO_SUSCRIPCION", "TIPO_CONTRATO", "TIPO_NOTIFICACION",
    "TIPO_REUNION", "TIPO_LEGAL", "TIPO_ACCESO", "TIPO_PROPUESTA",
    "ESTADO_PENDIENTE", "ESTADO_ESPERANDO", "ESTADO_RESUELTO"
  ];

  LISTA_ETIQUETAS.forEach(NOMBRE => {
    try {
      if (!GmailApp.getUserLabelByName(NOMBRE)) {
        GmailApp.createLabel(NOMBRE);
        Logger.log("✅ CREADA: " + NOMBRE);
      } else {
        Logger.log("⚠️ YA EXISTE: " + NOMBRE);
      }
    } catch (E) {
      Logger.log("❌ ERROR CON " + NOMBRE + ": " + E.message);
    }
  });
}

// ─── SE EJECUTA CADA 5 MIN HASTA TERMINAR ──────────────────────
function CONTINUAR_CLASIFICACION() {
  const PROPS = PropertiesService.getScriptProperties();
  let INICIO = parseInt(PROPS.getProperty("INICIO") || "0");
  const LOTE = 100;

  const HILOS = GmailApp.search("in:anywhere", INICIO, LOTE);

  HILOS.forEach(HILO => {
    try {
      const MENSAJE = HILO.getMessages()[0];
      const TEXTO = (
        MENSAJE.getSubject() + " " +
        MENSAJE.getPlainBody().substring(0, 1000)
      ).toUpperCase();

      REGLAS.forEach(REGLA => {
        const COINCIDE = REGLA.KEYWORDS.some(KW => TEXTO.includes(KW));
        if (COINCIDE) {
          REGLA.ETIQUETAS.forEach(NOMBRE_ETIQUETA => {
            const LABEL = GmailApp.getUserLabelByName(NOMBRE_ETIQUETA);
            if (LABEL) LABEL.addToThread(HILO);
          });
        }
      });
    } catch (E) {
      Logger.log("⚠️ ERROR EN HILO: " + E.message);
    }
  });

  const NUEVO_INICIO = INICIO + HILOS.length;
  Logger.log(`📦 PROCESADOS: HILOS ${INICIO} → ${NUEVO_INICIO}`);

  if (HILOS.length < LOTE) {
    ScriptApp.getProjectTriggers().forEach(T => {
      if (T.getHandlerFunction() === "CONTINUAR_CLASIFICACION") {
        ScriptApp.deleteTrigger(T);
      }
    });
    PROPS.deleteProperty("INICIO");
    Logger.log("🎯 SISTEMA COMPLETADO. GMAIL CLASIFICADO AL 100%.");
  } else {
    PROPS.setProperty("INICIO", NUEVO_INICIO.toString());
    Logger.log("⏳ CONTINUARÁ EN 5 MIN DESDE EL HILO " + NUEVO_INICIO);
  }
}
```

## 🚀 PASOS PARA EJECUTAR

1. ABRIR **GOOGLE DRIVE → NUEVO → MÁS → GOOGLE APPS SCRIPT**
2. BORRAR EL CÓDIGO VACÍO Y PEGAR EL SCRIPT COMPLETO
3. GUARDAR CON **CTRL+S**
4. SELECCIONAR LA FUNCIÓN **`iniciarSistemaCompleto`**
5. PULSAR ▶️ **EJECUTAR**
6. ACEPTAR LOS PERMISOS QUE PIDA GOOGLE
7. ESPERAR 20-30 MINUTOS — EL SISTEMA TRABAJA SOLO

PROGRESO VISIBLE EN: **VER → REGISTROS DE EJECUCIÓN**
